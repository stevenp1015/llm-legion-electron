# RATIONALE FOR ACTION:

def preprocess_nfl_data(df: pd.DataFrame) -> pd.DataFrame:
    """
    Applies a series of undisputed, foundational preprocessing steps to the raw NFL dataset.
    """
    # STEP 1: ELIMINATE IMPORT ARTIFACTS
    # JUSTIFICATION: The columns 'Unnamed: 73', 'Unnamed: 74', and 'Unnamed: 75' are artifacts
    # of the data import process, containing over 99.9% null values. They hold no signal and
    # represent pure noise. Their removal is a non-negotiable act of data hygiene that reduces
    # memory footprint and simplifies all subsequent analysis. This is the most parsimonious action.
    cols_to_drop = ['Unnamed: 73', 'Unnamed: 74', 'Unnamed: 75']
    df_clean = df.drop(columns=cols_to_drop)
    print(f"Initial columns: {len(df.columns)}. Columns after dropping artifacts: {len(df_clean.columns)}.")

    # STEP 2: ENGINEER THE PRIMARY BINARY TARGET
    # JUSTIFICATION: As per my Doctrine of Sequenced Complexity, our initial focus must be on the
    # primary win/loss question. Before we can model, we must define the ground truth. This step
    # creates the 'home_team_win' binary variable. It includes a specific, data-driven check for
    # ties, which are a known edge case. Handling them explicitly (by coding them as a non-win, '0')
    # is intellectually superior to ignoring them or allowing them to pollute the target.
    ties = df_clean[df_clean['score_home'] == df_clean['score_away']].shape[0]
    print(f"Data verification: Found {ties} tie games in the dataset.")
    df_clean['home_team_win'] = (df_clean['score_home'] > df_clean['score_away']).astype(int)
    print("Engineered binary target 'home_team_win'.")

    # STEP 3: TRANSFORM TIME-OF-POSSESSION INTO A NUMERICAL FEATURE
    # JUSTIFICATION: The 'possession_home' and 'possession_away' columns are stored as 'MM:SS'
    # strings, which are useless for quantitative analysis. The only defensible path is to convert
    # this representation into a continuous numerical variable—total seconds. This transformation
    # unlocks the feature's potential predictive power. An alternative, like splitting into two
    # integer columns, would unnecessarily increase dimensionality without providing superior signal.
    def convert_possession_to_seconds(possession_str):
        if isinstance(possession_str, str):
            try:
                minutes, seconds = map(int, possession_str.split(':'))
                return minutes * 60 + seconds
            except ValueError:
                return np.nan
        return np.nan

    df_clean['possession_home_seconds'] = df_clean['possession_home'].apply(convert_possession_to_seconds)
    df_clean['possession_away_seconds'] = df_clean['possession_away'].apply(convert_possession_to_seconds)
    df_clean = df_clean.drop(columns=['possession_home', 'possession_away'])
    print("Transformed 'possession' columns into numerical 'possession_seconds' and dropped originals.")
    
    return df_clean

This script is the implementation of the three-universe experiment mandated by the Commander

Universe A (The Vetoed Hypothesis): ties_as_loss. This dataset encodes ties as 0, lumping them

but potentially conflates two different signals (a loss and a tie) into one outcome.
Universe B (The Exclusion Hypothesis): no_ties. This dataset completely removes all games that
resulted in a tie. The rationale is to train the model on a 'pure' set of win/loss outcomes,
avoiding the ambiguity of a tie. The risk is information destruction; we are blinding the model
to a real-world, albeit rare, outcome.
Universe C (The Multi-Class Hypothesis): multi_class. This dataset reframes the problem from
binary to multi-class classification. The target is encoded as Win (2), Tie (1), or Loss (0).
The rationale is that this preserves the maximum amount of information about the outcome. The risk
is introducing a massively imbalanced class (ties) that may be difficult for a model to learn effectively.
By training an identical baseline model on each of these three universes, we will replace assumption
with data. The performance metrics from this trial will form the intellectually defensible basis
for our final choice of target variable encoding.
